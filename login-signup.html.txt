<script>
    function showSignup() {
        document.getElementById("loginForm").classList.remove("active");
        document.getElementById("bookingForm").classList.remove("active");
        document.getElementById("welcomeForm").classList.remove("active");
        document.getElementById("signupForm").classList.add("active");
    }

    function showLogin() {
        document.getElementById("signupForm").classList.remove("active");
        document.getElementById("bookingForm").classList.remove("active");
        document.getElementById("welcomeForm").classList.remove("active");
        document.getElementById("loginForm").classList.add("active");
    }

    function showBooking() {
        document.getElementById("loginForm").classList.remove("active");
        document.getElementById("signupForm").classList.remove("active");
        document.getElementById("welcomeForm").classList.remove("active");
        document.getElementById("bookingForm").classList.add("active");
    }

    function showWelcome() {
        document.getElementById("loginForm").classList.remove("active");
        document.getElementById("signupForm").classList.remove("active");
        document.getElementById("bookingForm").classList.remove("active");
        document.getElementById("welcomeForm").classList.add("active");
    }
</script>
